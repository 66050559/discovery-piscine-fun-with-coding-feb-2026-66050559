<script>
function calculate() {
    let a = document.getElementById("left").value;
    let b = document.getElementById("right").value;
    let op = document.getElementById("op").value;
    let display = document.getElementById("display-result");

    // ตรวจสอบว่าเป็นตัวเลขบวกเท่านั้นตาม logic เดิมของคุณ
    if (!/^\d+$/.test(a) || !/^\d+$/.test(b)) {
        alert("Error :(");
        display.innerText = "Error :(";
        return;
    }

    a = parseInt(a);
    b = parseInt(b);
    let result;

    if ((op === "/" || op === "%") && b === 0) {
        alert("It’s over 9000!");
        console.log("It’s over 9000!");
        display.innerText = "It’s over 9000!";
        return;
    }

    if (op === "+") result = a + b;
    else if (op === "-") result = a - b;
    else if (op === "*") result = a * b;
    else if (op === "/") result = a / b;
    else if (op === "%") result = a % b;

    alert(result);
    console.log(result);
    display.innerText = "Result: " + result;
}
